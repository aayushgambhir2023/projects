import matplotlib.pyplot as plt
import numpy as np

def cos_wave (A, f, phi, x_axis):
    y_n= A*np.cos(2*np.pi*f*x_axis-phi) #Calculating the values
    return y_n
phi_rad = (np.pi/180)*40 #converting from degrees to radians 
x_axis = np.arange (0,2,0.001) #time with a time step of 0.001
plt.figure(figsize=(8, 5))
plt.plot(x_axis, cos_wave(10,1,phi_rad,x_axis), 'r:') #for frequency 1 = 1 Hz
plt.plot(x_axis, cos_wave (10,2,phi_rad,x_axis), 'k-') #for frequency 2 = 2 Hz
plt.legend(['f1 = 1 Hz', 'f2 = 2 Hz']) #creating the legend
plt.xlabel('Time (s)')
plt.ylabel('y_n') 
plt.grid()
plt.show()

import matplotlib.pyplot as plt
import numpy as np

def lin_comb (A1,A2,f1, f2, phi1,phi2, x_axis):
    v_t = A1*np.sin(2*np.pi*f1*x_axis-phi1)+A2*np.cos(2*np.pi*f2*x_axis-phi2)
    return v_t
phi_rad1 = (np.pi/180)*45 
phi_rad2 = (np.pi/180)*75
x_axis = np.arange (0,np.pi,0.001)
plt.figure(figsize=(8, 5))
plt.plot(x_axis, lin_comb(5,-5,1,3,phi_rad1,phi_rad2,x_axis), 'k-') #for frequency 1 = 1 Hz
plt.xlabel('Time')
plt.ylabel('v_t(t)')

import matplotlib.pyplot as plt 
import numpy as np
def current (c, r, v, time):
    current_time = v*(1-np.exp(-time/(r*c))) #calculating the current 
    return current_time
def voltage (c, r, v, time):
    voltage_time = (v/r)*np.exp(-time/(r*c)) #calculating the voltage
    return voltage_time

time = np.linspace(0,0.04,1000) #time with appopriate time step
plt.figure(figsize=(8, 5))
plt.plot (time, current(0.002,1,5,time)) 
plt.plot(time, voltage(0.002,1,5,time))
plt.legend(['v(t)', 'i(t)'])
plt.title('RC Circuit') #title 
plt.xlabel('Time(s)')
plt.ylabel('Voltage/Current') 
plt.show()
